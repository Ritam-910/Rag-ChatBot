   ```bash
   git clone https://github.com/Ritam-910/Rag-ChatBot.git
   cd Rag-ChatBot
   ```

   ```bash
   python -m venv venv
   ```

   ```bash
   pip install -r requirements.txt
   ```

   ```bash
   streamlit run app.py
   ```
